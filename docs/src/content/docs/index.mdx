  <script is:inline>{`
    (function() {
      var wrap = document.querySelector('.cb-rotate-wrap');
      var rotate = document.querySelector('.cb-rotate');
      var words = document.querySelectorAll('.cb-rotate > span');
      if (!wrap || !rotate || !words.length) return;
      // The container is whatever block-level element the wrap lives inside
      // (the hero lead paragraph). Use it for prefix/suffix measurement and
      // as the target of .cb-headline-stacked (renamed .cb-lead-stacked).
      var headline = wrap.parentElement;
      if (!headline) return;

      // words[last] is a duplicate of words[0] so the scroll animation has
      // a smooth target before snapping back. Only cycle through 0..n-1.
      var n = words.length - 1;
      var i = 0;

      // Helper: build a hidden measurement span that inherits headline font
      function makeTest() {
        var t = document.createElement('span');
        t.style.cssText = 'position:absolute;visibility:hidden;white-space:nowrap;font:inherit;';
        return t;
      }

      // Measure the widest word (used as the locked wrap width — no more
      // width animation per tick means no layout shifts).
      function measureMaxWord() {
        var test = makeTest();
        test.className = wrap.className; // inherit .cb-rotate-wrap font/gradient
        headline.appendChild(test);
        var max = 0;
        for (var j = 0; j <= n; j++) {
          test.textContent = words[j].textContent;
          var w = test.getBoundingClientRect().width;
          if (w > max) max = w;
        }
        headline.removeChild(test);
        return Math.ceil(max) + 8; // fudge for kerning + rounding
      }

      // Lock the wrap width to the widest word so tick() never has to
      // change width and the line never reflows within the group. Natural
      // text wrapping (plus white-space:nowrap on .cb-rotate-group)
      // handles layout at narrow viewports: the group stays glued as a
      // unit and breaks as a whole when there's no room.
      function applyLayout() {
        var maxW = measureMaxWord();
        wrap.style.width = maxW + 'px';
      }

      // Rotate one word forward. Pure transform — no width change.
      function tick() {
        i = (i + 1) % (n + 1);
        rotate.style.transform = 'translateY(-' + (i * 1.15) + 'em)';
        if (i === n) {
          // Landed on the duplicate word — snap back to index 0 without
          // animation so the next tick moves smoothly forward again.
          setTimeout(function() {
            rotate.style.transition = 'none';
            rotate.style.transform = 'translateY(0)';
            i = 0;
            void rotate.offsetHeight; // force reflow
            rotate.style.transition = '';
          }, 550);
        }
      }

      function start() {
        requestAnimationFrame(function() {
          applyLayout();
          setInterval(tick, 2500);
        });
      }

      var resizeTimer = null;
      window.addEventListener('resize', function() {
        clearTimeout(resizeTimer);
        resizeTimer = setTimeout(applyLayout, 150);
      });

      if (document.fonts && document.fonts.ready) {
        document.fonts.ready.then(start);
      } else {
        window.addEventListener('load', start);
      }
    })();
  `}</script>
