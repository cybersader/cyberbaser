    (function() {
      var wrap = document.querySelector('.cb-rotate-wrap');
      var rotate = document.querySelector('.cb-rotate');
      var words = document.querySelectorAll('.cb-rotate > span');
      if (!wrap || !rotate || !words.length) return;
      var n = words.length - 1;
      var i = 0;
      var widths = [];

      function setWidths() {
        var test = document.createElement('span');
        test.style.cssText = 'position:absolute;visibility:hidden;white-space:nowrap;';
        test.className = wrap.className;
        wrap.parentNode.appendChild(test);
        var newWidths = [];
        for (var j = 0; j <= n; j++) {
          test.textContent = words[j].textContent;
          newWidths.push(Math.ceil(test.getBoundingClientRect().width) + 8);
        }
        wrap.parentNode.removeChild(test);
        widths = newWidths;
        wrap.style.width = widths[i] + 'px';
      }

      function tick() {
        i = (i + 1) % (n + 1);
        wrap.style.width = widths[i] + 'px';
        rotate.style.transform = 'translateY(-' + (i * 1.15) + 'em)';
        if (i === n) {
          setTimeout(function() {
            rotate.style.transition = 'none';
            wrap.style.transition = 'none';
            rotate.style.transform = 'translateY(0)';
            wrap.style.width = widths[0] + 'px';
            i = 0;
            void rotate.offsetHeight;
            rotate.style.transition = '';
            wrap.style.transition = '';
          }, 550);
        }
      }

      function start() {
        requestAnimationFrame(function() {
          setWidths();
          setInterval(tick, 2500);
        });
      }

      var resizeTimer = null;
      window.addEventListener('resize', function() {
        clearTimeout(resizeTimer);
        resizeTimer = setTimeout(setWidths, 150);
      });

      if (document.fonts && document.fonts.ready) {
        document.fonts.ready.then(start);
      } else {
        window.addEventListener('load', start);
      }
    })();
  `}</script>
